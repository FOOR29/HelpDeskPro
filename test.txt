// app/api/tickets/[id]/route.ts

import { NextResponse } from "next/server"
import { auth } from "@/src/auth"
import { db } from "@/src/lib/db"

type Params = { params: { id: string } }

export async function GET(_req: Request, { params }: Params) {
    try {
        const ticket = await db.ticket.findUnique({
            where: { id: params.id },
            include: {
                createdBy: { select: { id: true, name: true, email: true } },
                assignedTo: { select: { id: true, name: true, email: true } },
                comments: {
                    orderBy: { createdAt: "asc" },
                    include: {
                        author: { select: { id: true, name: true, email: true, role: true } },
                    },
                },
            },
        })
        if (!ticket) {
            return new NextResponse("Ticket not found", { status: 404 })
        }
        return NextResponse.json(ticket)
    } catch (error) {
        console.error("[TICKET_GET]", error)
        return new NextResponse("Internal error", { status: 500 })
    }
}

export async function PATCH(req: Request, { params }: Params) {
    try {
        const session = await auth()
        if (!session || !session.user) {
            return new NextResponse("Unauthorized", { status: 401 })
        }

        const role = session.user.role as "client" | "agent"
        if (role !== "agent") {
            // solo agentes pueden cambiar estado/prioridad/asignaci√≥n
            return new NextResponse("Forbidden", { status: 403 })
        }

        const body = await req.json()
        const { status, priority, assignedToId } = body as {
            status?: "OPEN" | "IN_PROGRESS" | "RESOLVED" | "CLOSED"
            priority?: "LOW" | "MEDIUM" | "HIGH"
            assignedToId?: string | null
        }

        // opcional: validar que el agente asignado exista y sea agent
        let data: any = {}
        if (status) data.status = status
        if (priority) data.priority = priority
        if (typeof assignedToId !== "undefined") {
            if (assignedToId === null || assignedToId === "") {
                data.assignedToId = null
            } else {
                const agent = await db.user.findUnique({
                    where: { id: assignedToId },
                })
                if (!agent || agent.role !== "agent") {
                    return new NextResponse("Invalid agent", { status: 400 })
                }
                data.assignedToId = assignedToId
            }
        }

        const updated = await db.ticket.update({
            where: { id: params.id },
            data,
        })

        return NextResponse.json(updated)
    } catch (error) {
        console.error("[TICKET_PATCH]", error)
        return new NextResponse("Internal error", { status: 500 })
    }
}
